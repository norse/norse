[docs]def test_tonic_spiking():
    p, s = izk.tonic_spiking
    ts = np.arange(0, 100, 0.25)
    T1 = ts[-1] / 10
    vs = []
    us = []
    cs = []

    for t in ts:
        vs.append(s.v)
        us.append(s.u)
        if t > T1:
            input_current = 14 * torch.ones(1)
        else:
            input_current = 0 * torch.ones(1)
        _, s = izhikevich_step(input_current, s, p)
        cs.append(input_current)
    return vs, cs
